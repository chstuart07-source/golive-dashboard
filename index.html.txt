<script>
const API = "/api/asana";
const PROJECT_ID = "1204989172185092";
const PKG_FIELD = "1210696618096750";
const GL_FIELD = "1211590050412697";
const VP_FIELD = "1204990658751105";
const DEBOOK_GID = "1209225331518753";
const SECTIONS = {"March 2026":"1212091088978746","April 2026":"1212091088978747","May 2026":"1212091088978748"};

const NC = [
  {k:"nc1",l:"Kick off call completed",w:8},
  {k:"nc2",l:"SOP shared",w:6},
  {k:"nc3",l:"KYC form completed",w:9},
  {k:"nc4",l:"Confirmed payment method adoption vs HubSpot",w:9},
  {k:"nc5",l:"All accounts connected to Stripe",w:10},
  {k:"nc6",l:"# of Restricted Soon resolved (any reason)",w:8},
  {k:"nc7",l:"Portal configured",w:9},
  {k:"nc8",l:"Non-rack pricing assigned",w:7},
  {k:"nc9",l:"Double check portal config upon launch",w:8},
  {k:"nc10",l:"Monitor payment activity post portal launch",w:7},
];
const B2B = [
  {k:"b1",l:"Sales → Imp. handoff call",w:8},
  {k:"b2",l:"Confirmed payment method adoption vs HubSpot",w:9},
  {k:"b3",l:"Transition date identified",w:8},
  {k:"b4",l:"SOP shared",w:6},
  {k:"b5",l:"KYC form completed",w:9},
  {k:"b6",l:"Tags received",w:7},
  {k:"b7",l:"Final review of comms template",w:7},
  {k:"b8",l:"Comms XN built",w:8},
  {k:"b9",l:"Comms sent",w:9},
  {k:"b10",l:"All accounts connected to Stripe",w:10},
  {k:"b11",l:"# of Restricted Soon resolved (any reason)",w:8},
  {k:"b12",l:"Portal configured",w:9},
  {k:"b13",l:"Non-rack pricing assigned",w:7},
  {k:"b14",l:"Double check portal config upon launch",w:8},
  {k:"b15",l:"Monitor payment activity post launch",w:7},
];

let clients=[],filterType="All",cur=null,subs=[],wtab="nc";
let W={nc:Object.fromEntries(NC.map(t=>[t.k,t.w])),b2b:Object.fromEntries(B2B.map(t=>[t.k,t.w]))};

async function asana(path,method,body){
  const r=await fetch(API,{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({path,method:method||"GET",body:body||null})
  });
  return r.json();
}

async function init(){
  document.getElementById("list").innerHTML='<div class="loading">Loading clients from Asana...</div>';
  clients=[];
  for(const[name,sid] of Object.entries(SECTIONS)){
    const d=await asana(`/sections/${sid}/tasks?opt_fields=gid,name,completed,custom_fields,subtasks`);
    if(!d.data) continue;
    for(const t of d.data){
      if(t.completed) continue;
      const pkg=(t.custom_fields||[]).find(f=>f.gid===PKG_FIELD);
      const gl=(t.custom_fields||[]).find(f=>f.gid===GL_FIELD);
      const vp=(t.custom_fields||[]).find(f=>f.gid===VP_FIELD);
      clients.push({
        gid:t.gid,name:t.name,
        isB2B:pkg?.enum_value?.name==="B2B",
        isDebook:vp?.enum_value?.gid===DEBOOK_GID,
        goLiveDate:gl?.date_value?.date||null,
        vpStatus:vp?.enum_value?.name||null,
        subtaskCount:(t.subtasks||[]).length,
        section:name
      });
    }
  }
  renderList();
}

function days(d){if(!d)return null;return Math.ceil((new Date(d)-new Date())/86400000)}
function um(d){if(d===null)return 1;if(d<=0)return 2;if(d<=7)return 1.8;if(d<=14)return 1.5;if(d<=30)return 1.2;return 1}
function score(tasks,subs,gl,wkey){
  if(!subs||!subs.length)return null;
  const m=um(days(gl));let tot=0,earn=0;
  for(const t of tasks){const w=(W[wkey][t.k]||t.w)*m;const s=subs.find(s=>s.name===t.l);tot+=w;if(s?.completed)earn+=w;}
  return tot===0?0:Math.round(earn/tot*100);
}
function col(s){
  if(s===null)return{bar:"#b4b2a9",txt:"#5f5e5a",bg:"#f1efe8"};
  if(s>=80)return{bar:"#639922",txt:"#3b6d11",bg:"#eaf3de"};
  if(s>=50)return{bar:"#ef9f27",txt:"#854f0b",bg:"#faeeda"};
  return{bar:"#e24b4a",txt:"#a32d2d",bg:"#fcebeb"};
}
function slabel(s){if(s===null)return"Not set up";if(s>=80)return"On track";if(s>=50)return"At risk";return"Critical"}
function dlabel(d){if(d===null)return"No date set";if(d<0)return`${Math.abs(d)}d overdue`;if(d===0)return"Go-live today";return`${d}d to go-live`}

function setF(f){
  filterType=f;
  ["All","New","B2B"].forEach(x=>document.getElementById("f-"+x).classList.toggle("active",x===f));
  renderList();
}

function renderList(){
  const q=document.getElementById("srch").value.toLowerCase();
  const list=clients.filter(c=>{
    if(filterType==="B2B"&&!c.isB2B)return false;
    if(filterType==="New"&&c.isB2B)return false;
    if(q&&!c.name.toLowerCase().includes(q))return false;
    return true;
  }).sort((a,b)=>(days(a.goLiveDate)??9999)-(days(b.goLiveDate)??9999));

  if(!list.length){document.getElementById("list").innerHTML='<div class="loading">No clients found.</div>';return;}
  document.getElementById("list").innerHTML=list.map(c=>`
    <div class="card${c.isDebook?" debook":""}" onclick="openClient('${c.gid}')">
      <div style="display:flex;align-items:center;gap:12px">
        <div style="flex:1">
          <div class="card-name">${c.name}${c.isDebook?'<span class="badge deb">Debook</span>':''}</div>
          <div class="card-meta">
            <span class="badge ${c.isB2B?"b2b":"newc"}">${c.isB2B?"B2B":"New"}</span>
            <span>${dlabel(days(c.goLiveDate))}</span>
            ${c.vpStatus?`<span style="color:#bbb">· ${c.vpStatus}</span>`:""}
          </div>
        </div>
        <div class="hint${c.subtaskCount?"":" warn"}">${c.subtaskCount?c.subtaskCount+" subtasks":"No checklist"}</div>
      </div>
    </div>`).join("");
}

async function openClient(gid){
  cur=clients.find(c=>c.gid===gid);
  document.getElementById("dash").style.display="none";
  document.getElementById("det").style.display="block";
  document.getElementById("detc").innerHTML='<div class="loading">Loading checklist from Asana...</div>';
  const d=await asana(`/tasks/${gid}/subtasks?opt_fields=gid,name,completed`);
  subs=d.data||[];
  renderDet();
}

function renderDet(){
  const c=cur,tasks=c.isB2B?B2B:NC,wkey=c.isB2B?"b2b":"nc";
  const sc=score(tasks,subs,c.goLiveDate,wkey),cl=col(sc);
  const done=subs.filter(s=>s.completed).length,has=subs.length>0,d=days(c.goLiveDate);
  document.getElementById("detc").innerHTML=`
    <div class="dhead">
      <div style="display:flex;justify-content:space-between;align-items:flex-start;margin-bottom:14px">
        <div>
          <div style="font-size:18px;font-weight:600">${c.name}</div>
          <div class="card-meta" style="margin-top:4px">
            <span class="badge ${c.isB2B?"b2b":"newc"}">${c.isB2B?"B2B":"New client"}</span>
            ${c.isDebook?'<span class="badge deb">⚑ Debook deal</span>':""}
            <span>${dlabel(d)}</span>
            ${c.vpStatus?`<span style="color:#bbb">· ${c.vpStatus}</span>`:""}
          </div>
        </div>
      </div>
      <div class="mgrid">
        <div class="metric" style="background:${cl.bg}"><label>Health score</label><span style="color:${cl.txt}">${sc!==null?sc+"%":"—"}</span></div>
        <div class="metric" style="background:#f5f5f3"><label>Tasks done</label><span>${has?done+"/"+tasks.length:"—"}</span></div>
        <div class="metric" style="background:${cl.bg}"><label>Status</label><span style="color:${cl.txt};font-size:15px">${slabel(sc)}</span></div>
      </div>
      ${sc!==null?`<div class="prog-bg"><div class="prog-fill" style="width:${sc}%;background:${cl.bar}"></div></div>`:""}
    </div>
    <div class="clist">
      <h3>Checklist</h3>
      ${!has?`<div class="setup"><p>No checklist subtasks found in Asana for this client.</p><button class="btn" onclick="setupCL()">Set up checklist in Asana</button></div>`
      :tasks.map(t=>{
        const s=subs.find(s=>s.name===t.l),chk=s?.completed||false,wv=W[wkey][t.k]||t.w;
        return`<div class="trow" onclick="toggle('${s?.gid||""}',${chk})">
          <div class="chk${chk?" on":""}">${chk?'<svg width="11" height="9" viewBox="0 0 11 9" fill="none"><path d="M1 4.5L4 7.5L10 1" stroke="white" stroke-width="1.8" stroke-linecap="round" stroke-linejoin="round"/></svg>':""}</div>
          <div class="tlabel${chk?" done":""}">${t.l}</div>
          <div class="twt">wt: ${wv}</div>
        </div>`;}).join("")}
    </div>`;
}

async function setupCL(){
  const tasks=cur.isB2B?B2B:NC;
  const btn=document.querySelector(".setup .btn");
  if(btn){btn.disabled=true;btn.textContent="Creating subtasks in Asana...";}
  for(const t of tasks){
    await asana("/tasks","POST",{data:{name:t.l,parent:cur.gid}});
  }
  const d=await asana(`/tasks/${cur.gid}/subtasks?opt_fields=gid,name,completed`);
  subs=d.data||[];
  cur.subtaskCount=subs.length;
  renderDet();
}

async function toggle(gid,was){
  if(!gid)return;
  document.querySelectorAll(".trow").forEach(r=>r.classList.add("busy"));
  await asana(`/tasks/${gid}`,"PUT",{data:{completed:!was}});
  subs=subs.map(s=>s.gid===gid?{...s,completed:!was}:s);
  renderDet();
}

function goBack(){
  document.getElementById("det").style.display="none";
  document.getElementById("dash").style.display="block";
  cur=null;subs=[];
}

function toggleW(){
  const p=document.getElementById("wp");
  p.style.display=p.style.display==="none"?"block":"none";
  renderW();
}
function switchWT(t){
  wtab=t;
  document.getElementById("wt-nc").classList.toggle("active",t==="nc");
  document.getElementById("wt-b2b").classList.toggle("active",t==="b2b");
  renderW();
}
function renderW(){
  const tasks=wtab==="nc"?NC:B2B;
  document.getElementById("wrows").innerHTML=tasks.map(t=>`
    <div class="wrow">
      <span>${t.l}</span>
      <input type="range" min="1" max="10" step="1" value="${W[wtab][t.k]||t.w}" oninput="upW('${wtab}','${t.k}',this.value,this.nextElementSibling)"/>
      <span class="wval">${W[wtab][t.k]||t.w}</span>
    </div>`).join("");
}
function upW(tab,k,v,el){W[tab][k]=+v;el.textContent=v;}

init();
